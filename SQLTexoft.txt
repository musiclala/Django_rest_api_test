CREATE DATABASE texoft;

create sequence category_seq
increment 1
start 1;

create sequence product_seq
increment 1
start 1;


create table Category (
	id integer Primary key DEFAULT nextval('category_seq'),
	title varchar(50) not null unique,
	description text
);

create table Product (
	id integer PRIMARY KEY DEFAULT nextval('product_seq'),
	title varchar(50) NOT NULL unique,
	description text,
	price double precision NOT NULL default 0 CONSTRAINT positive_price check (price > 0),
	count numeric NOT NULL default 0 CONSTRAINT positive_count check (count > 0),
	category_id integer REFERENCES Category(id)
);
